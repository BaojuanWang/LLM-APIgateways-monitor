// 在 React 加载前立即从 localStorage 读取设置，避免闪烁 (function() { try { var name = null; var logo = null; // 优先从 status 缓存读取 var statusStr = localStorage.getItem('status'); if (statusStr) { try { var s = JSON.parse(statusStr); if (s && s.system_name) name = s.system_name; if (s && s.logo) logo = s.logo; } catch(e) {} } // 备选：从 system-config-storage 缓存读取 if (!name || !logo) { var configStr = localStorage.getItem('system-config-storage'); if (configStr) { try { var c = JSON.parse(configStr); // Skip old default names from zustand cache if (!name && c && c.config && c.config.systemName && c.config.systemName !== 'New API' && c.config.systemName !== 'EASY API') name = c.config.systemName; if (!logo && c && c.config && c.config.logo) logo = c.config.logo; } catch(e) {} } } // 应用标题 if (name) { document.title = name; var metaTitle = document.querySelector('meta[name="title"]'); if (metaTitle) metaTitle.setAttribute('content', name); } // 应用 favicon if (logo) { var existing = document.querySelectorAll('link[rel~="icon"]'); existing.forEach(function(l) { l.remove(); }); var link = document.createElement('link'); link.rel = 'icon'; link.href = logo; document.head.appendChild(link); } } catch(e) {} })();
